Pattern p = Pattern.compile("([ab])\\1");

Pattern p = Pattern.compile("([ab])\\1");
Matcher m = p.matcher("aabbcc");
m.matches(); /* false */

m.find(); /* true */
m.start(); /* 0 */
m.end(); /* 2 */
m.find(); /* true */
m.start(); /* 2 */
m.end(); /* 4 */
m.find(); /* false */

p.split("aabbcc"); /* ["", "", "cc"] */
p.splitWithDelimiters("aabbcc",0); /* ["", "aa", "", "bb", "cc"] */
----
